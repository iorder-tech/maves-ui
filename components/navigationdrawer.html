NavigationDrawer #

The m-navigation-drawer component is what your users will utilize to navigate through the application. The navigation-drawer is pre-configured to work with or without vue-router right out the box.

Examples #

<template>
	<m-card style="position: relative; overflow: hidden; height: 500px">
		<m-button @click="drawer = true">Open Drawer</m-button>
		<m-navigation-drawer v-model="drawer" location="left">
			Test
		</m-navigation-drawer>
	</m-card>
</template>

<script setup lang="ts">
	import { ref, Ref } from 'vue'

	const drawer: Ref<boolean> = ref(false)
</script>
